OPEN SYMMETRIC KEY BAMKEY192 DECRYPTION BY CERTIFICATE BOOKSANDMOVIESCERTIFICATE
OPEN SYMMETRIC KEY BAMKEY256 DECRYPTION BY CERTIFICATE BOOKSANDMOVIESCERTIFICATE

SELECT CONVERT(nvarchar, DecryptByKey(NOM_CRYPT)) AS NOM, 
	CONVERT(nvarchar, DecryptByKey(PRENOM_CRYPT)) AS PRENOM,
	CONVERT(nvarchar, DecryptByKey(ADRESSEMAIL_CRYPT)) AS ADRESSEMAIL,
	CONVERT(varchar, DecryptByKey(MOTDEPASSE_CRYPT)) AS MOTDEPASSE,
	CONVERT(varchar, DecryptByKey(DATENAISSANCE_CRYPT)) AS DATENAISSANCE
FROM UTILISATEUR

CLOSE SYMMETRIC KEY BAMKEY256
CLOSE SYMMETRIC KEY BAMKEY192
